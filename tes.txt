public String calculateGrade() {
        if (finalGrade >= 90) {
            return "A";
        } else if (finalGrade >= 80) {
            return "C";
        } else if (finalGrade >= 70) {
            return "B";
        } else if (finalGrade >= 60) {
            return "D";
        } else {
            return "F"; } }
